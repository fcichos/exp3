def aperture(r,z=10e-6,R=5e-6,wl=532e-9):

    theta=np.arctan(np.sqrt(r**2)/z)
    x=2*np.pi*R*np.sin(theta)/wl

    return((2*sp.j1(x)/x)**2)

def caperture(r,z=10e-6,R=5e-6,wl=532e-9):

    theta=np.arctan(np.sqrt(r[0]**2+r[1]**2)/z)
    x=2*np.pi*R*np.sin(theta)/wl

    return((2*sp.j1(x)/x)**2)
